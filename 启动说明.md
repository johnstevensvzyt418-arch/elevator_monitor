python -c "
import paho.mqtt.client as mqtt, time, urllib.request, urllib.parse

c = mqtt.Client(client_id='elevator-bridge', protocol=mqtt.MQTTv5, callback_api_version=mqtt.CallbackAPIVersion.VERSION2)
c.username_pw_set('admin', 'SZTUbdi@1005')

def on_connect(client, userdata, flags, rc, props):
    print(f'[Bridge] Connected: rc={rc}')
    client.subscribe('/Elevator', qos=1)
    print('[Bridge] Subscribed to /Elevator')

def on_message(client, userdata, msg):
    payload = msg.payload.decode().strip()
    # 打印完整报文 + 长度（便于排查协议问题）
    print(f'[Bridge] RECEIVED ({len(payload)}chars): {payload}')
    
    # 从报文中正确提取 deviceId (偏移21-28) 和设备时间 (偏移12-19)
    device_id = payload[21:29] if len(payload) >= 29 else 'unknown'
    device_time = payload[12:20] if len(payload) >= 20 else time.strftime('%H:%M:%S')
    
    data = urllib.parse.urlencode({'data': payload, 'time': device_time, 'elevatorID': device_id}).encode()
    try:
        resp = urllib.request.urlopen(urllib.request.Request('http://localhost:10008/api/v2/mnk', data=data, method='POST'), timeout=5)
        result = resp.read().decode()
        print(f'[Bridge] device={device_id} → HTTP {result}')
    except Exception as e:
        print(f'[Bridge] device={device_id} → HTTP ERROR: {e}')

c.on_connect = on_connect
c.on_message = on_message
c.connect('tcp.sealosbja.site', 35205, 60)
print('[Bridge] Connecting to tcp.sealosbja.site:35205 ...')
c.loop_forever()
"
